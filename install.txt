sudo apt update
sudo apt install qemu-system
qemu-system-x86_64 --version

// gcc build dependencies (to build gcc cross compiler)
sudo apt install bison flex libgmp3-dev libmpc-dev libmpfr-dev texinfo

// -- building cross compilerma
export PREFIX="$HOME/opt/cross"
export TARGET=x86_64-elf
export PATH="$PREFIX/bin:$PATH"

//install binutils 
//you'll need to get binaries from https://ftp.gnu.org/gnu/binutils/binutils-2.44.tar.gz . unzip it into $HOME/src
cd $HOME/src
mkdir build-binutils
cd build-binutils
../binutils-2.44/configure --target=$TARGET --prefix="$PREFIX" --with-sysroot --disable-nls --disable-werror
make
make install

//install gcc (target is x86_64, so make sure to turn of 'red zone')
//https://ftp.gnu.org/gnu/gcc/gcc-12.5.0/gcc-12.5.0.tar.gz , unzip into $HOME/src
cd $HOME/src
mkdir build-gcc
../gdb.x.y.z/configure --target=$TARGET --prefix="$PREFIX" --disable-werror
make all-gdb
make install-gdb
